### Schema 推測 `provisional_providers`（PRD 沒給細節，需 Leo 跟 Kristine 確認）
```prisma
model provisional_providers {
  id            String   @id @default(cuid())
  npi           String   @db.VarChar(20)
  practice_id   Int      // FK to ?
  provider_name String?
  first_name    String?
  last_name     String?
  detected_from String   // 'HL7' | 'PORTAL'
  detected_at   DateTime @default(now())
  status        String   @default("PENDING") // PENDING | CONVERTED | REJECTED
  source_message_id String? // HL7 message that triggered
  notes         String?  @db.Text
  created_at    DateTime @default(now())
  updated_at    DateTime @updatedAt
  @@unique([npi, practice_id])
  @@index([practice_id, status])
}
```
